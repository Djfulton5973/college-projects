47 void Azul::moveForward() {
48 float speedX = 0.0f, speedY = 0.0f;
49 sf::Vector2i moveTo;
50 int moveRow = m_row;
51 int moveColumn = m_column;
52 if (!frontIsBlocked()) {
53 switch (m_facing) {
54 case North:
55 moveRow -= 1;
56 speedY = -1;
57 break;
58 case East:
59 moveColumn += 1;
60 speedX = 1;
61 break;
62 case South:
63 moveRow += 1;
64 speedY = 1;
65 break;
66 case West:
67 moveColumn -= 1;
68 speedX = -1;
69 }
70
71 AzulCell* currentCell = m_world.getCell(m_row, m_column);
72 sf::Vector2i startFrom = currentCell->getCenter();
73 float size = currentCell->getSize();
74 float speed = m_world.getAnimationSpeed();
75 float distance = 0.0f;
76 while (m_world.runAnimation() && distance <= size) {
77 m_azul.setHorizontalSpeed(m_world.getAnimationSpeed() * speedX);
78 m_azul.setVerticalSpeed(m_world.getAnimationSpeed() * speedY);
79 sf::Vector2f current = m_azul.getPosition();
80 distance = std::sqrt(std::pow(startFrom.y - current.y, 2) + std::pow(startFrom.x - current.x,2));
81 }
82 m_azul.setHorizontalSpeed(0.0f);
83 m_azul.setVerticalSpeed(0.0f);
84 m_row = moveRow;
85 m_column = moveColumn;
86 currentCell = m_world.getCell(m_row, m_column);
87 sf::Vector2i newPosition = currentCell->getCenter();
88 m_azul.setPosition(newPosition.x, newPosition.y);
89 }

92 void Azul::turnLeft() {
93 fgcu::Direction newFacing;
94 switch (m_facing) {
95 case North:
96 newFacing = West;
97 break;
98 case East:
99 newFacing = North;
100 break;
101 case South:
102 newFacing = East;
103 break;
104 case West:
105 newFacing = South;
106 }
107 int origin = static_cast<int>(m_facing);
108 if (origin == 0)
109 origin = 360;
110 int target = static_cast<int>(newFacing);
111
112 while (m_world.runAnimation()) {
113 m_azul.setRotationalSpeed(m_world.getAnimationSpeed() * -1.0);
114
115 int heading = m_azul.getRotation();
116 if (!(heading >= target && heading <= origin))
117 break;
118 }
119 m_azul.setRotationalSpeed(0.0f);
120 m_azul.setRotation(target);
121 m_facing = newFacing;
122 }
